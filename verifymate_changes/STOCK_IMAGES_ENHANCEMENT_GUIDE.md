## Image Specifications

### Technical Requirements

- **Format:** JPEG (.jpg)
- **Minimum Resolution:** 1920x1080 pixels (Full HD)
- **Recommended Resolution:** 2560x1440 pixels (2K) or 3840x2160 pixels (4K)
- **Aspect Ratio:** 16:9 (landscape orientation)
- **File Size:** 200KB - 2MB (optimized for web)
- **Quality:** High quality, professional photography
- **Color Space:** sRGB

### Visual Requirements

- **Professional quality:** No amateur or low-quality images
- **Well-lit:** Proper exposure, no over/underexposure
- **Sharp focus:** Clear and crisp, no blur
- **Composition:** Well-framed, rule of thirds
- **Color:** Vibrant but natural colors
- **No watermarks:** Clean images without watermarks or logos
- **Modern aesthetic:** Contemporary style matching JournalMate brand

const https = require('https');

const path = require('path');

async function searchUnsplash(query) {
  return new Promise((resolve, reject) => {
    const options = {
      hostname: 'api.unsplash.com',
      path: `/search/photos?query=${encodeURIComponent(query)}&per_page=1&orientation=landscape`,
      headers: {
        'Authorization': `Client-ID ${UNSPLASH_ACCESS_KEY}`
      }
    };

    https.get(options, (res) => {
      let data = '';
      res.on('data', chunk => data += chunk);
      res.on('end', () => {
        try {
          const json = JSON.parse(data);
          if (json.results && json.results.length > 0) {
            resolve(json.results[0].urls.full);
          } else {
            reject(new Error('No results found'));
          }
        } catch (e) {
          reject(e);
        }
      });
    }).on('error', reject);
  });
}

async function downloadImage(url, filepath) {
  return new Promise((resolve, reject) => {
    const file = fs.createWriteStream(filepath);
    https.get(url, (res) => {
      res.pipe(file);
      file.on('finish', () => {
        file.close();
        resolve();
      });
    }).on('error', (err) => {
      fs.unlink(filepath, () => {}); // Delete incomplete file
      reject(err);
    });
  });
}

async function main() {
  if (!UNSPLASH_ACCESS_KEY) {
    console.error('ERROR: UNSPLASH_ACCESS_KEY environment variable not set');
    console.error('Get your API key from https://unsplash.com/developers');
    process.exit(1);
  }

  if (!fs.existsSync(OUTPUT_DIR)) {
    fs.mkdirSync(OUTPUT_DIR, { recursive: true });
  }

  console.log('Starting stock image download...\n');

  for (let i = 0; i < imageSpecs.length; i++) {
    const spec = imageSpecs[i];
    const filepath = path.join(OUTPUT_DIR, spec.filename);

    console.log(`[${i + 1}/${imageSpecs.length}] Downloading: ${spec.filename}`);
    console.log(`   Query: "${spec.query}"`);

    try {
      const imageUrl = await searchUnsplash(spec.query);
      await downloadImage(imageUrl, filepath);
      console.log(`   ✓ Success\n`);
    } catch (error) {
      console.error(`   ✗ Error: ${error.message}\n`);
    }

    // Rate limiting - wait 1 second between requests
    if (i < imageSpecs.length - 1) {
      await new Promise(resolve => setTimeout(resolve, 1000));
    }
  }

  console.log('Download complete!');
}

main().catch(console.error);

## Troubleshooting

**Build fails with image errors:**
- Check filenames match exactly
- Verify file extensions are .jpg (not .jpeg)
- Ensure files are not corrupted

**Images not displaying:**
- Clear browser cache
- Check browser console for errors
- Verify image paths in CommunityPlansPage.tsx

**Images look pixelated:**
- Ensure minimum 1920x1080 resolution
- Check JPEG quality is at least 85%
- Download higher resolution versions

**Images too large (slow loading):**
- Optimize with TinyJPG.com
- Target 200KB-500KB per image
- Use JPEG quality 80-85%
